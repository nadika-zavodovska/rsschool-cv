let weather = {
  paris: {
    temp: 19.7,
    humidity: 80
  },
  tokyo: {
    temp: 17.3,
    humidity: 50
  },
  lisbon: {
    temp: 30.2,
    humidity: 20
  },
  "san francisco": {
    temp: 20.9,
    humidity: 100
  },
  oslo: {
    temp: -5,
    humidity: 20
  }
};
let userCity = prompt("Enter a city?");
userCity = userCity.toLowerCase().trim();
if (weather[userCity] !== undefined) {
  let tempCelsius = Math.round(weather[userCity].temp);
  let tempFahrenheit = (tempCelsius * 9) / 5 + 32;
  alert(
    `It is currently ${Math.round(
      weather[userCity].temp
    )}°C (${tempFahrenheit}°F) in ${
      userCity.charAt(0).toUpperCase() + userCity.slice(1)
    } with a humidity of ${Math.round(weather[userCity].humidity)}%`
  );
} else {
  alert(
    `Sorry, we don't know the weather for this city, try 
    going to https://www.google.com/search?q=weather+${userCity}`
  );
}
